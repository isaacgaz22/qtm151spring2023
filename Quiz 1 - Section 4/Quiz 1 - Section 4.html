csv_files = ['data_quiz/wdi_co2_2005.csv', 'data_quiz/wdi_co2_2010.csv', 'data_quiz/wdi_co2_2015.csv']

index = 0

for i in range(len(csv_files)):
    dataset = pd.read_csv(csv_files[i])
    co2_gasfuel_kt = dataset["co2_gasfuel_kt"]
    co2_liquidfuel_kt = dataset["co2_liquidfuel_kt"]
    total_emis = co2_gasfuel_kt + co2_liquidfuel_kt

    plt.hist(total_emis, bins = 50)
    plt.xlabel('Total Emission')
    plt.ylabel('Frequency')
    plt.title("My Histogram for " + csv_files[i])
    plt.show()

    index += 1
